function handleFileUpload(files){
if(!files||!files.length)return;

/* Auto-open existing Files pop-up in compact clean view.
   Initial view shows only file name and progress percentage.
   User can expand using existing arrow button. */
if(fileDropdown){
fileDropdown.classList.add("active");
fileDropdown.classList.add("compact");
fileDropdown.classList.remove("body-collapsed");
}

const existingNames=new Set(
uploadedFiles
.map(f=>String(f.name||"").trim().toLowerCase())
.filter(Boolean)
);

fileQueue.forEach(q=>{
if(q&&q.fileMeta&&q.fileMeta.name){
existingNames.add(String(q.fileMeta.name).trim().toLowerCase());
}
});

let acceptedCount=0;
let skippedMessages=[];

files.forEach(file=>{
const fileName=String(file.name||"").trim();
const normalizedFileName=fileName.toLowerCase();
const ext=getExtension(fileName);

if(!["csv","xlsx","xls"].includes(ext)){
skippedMessages.push("Wrong file format skipped: "+fileName);
return;
}

if(existingNames.has(normalizedFileName)){
skippedMessages.push("Duplicate file skipped: "+fileName);
return;
}

if(file.size>CONFIG.MAX_FILE_SIZE_MB*1024*1024){
skippedMessages.push("File too large skipped: "+fileName);
return;
}

if(uploadedFiles.length+acceptedCount>=CONFIG.MAX_FILE_COUNT){
skippedMessages.push("File count limit exceeded. Skipped: "+fileName);
return;
}

/* Show all accepted files immediately before parsing starts */
const fileMeta={
id:"f_"+Date.now()+"_"+Math.random().toString(36).slice(2),
name:fileName,
type:ext,
size:formatSize(file.size),
rawSize:file.size,
status:"Queued",
monthYear:extractMonthYear(fileName),
rowCount:0,
progress:0
};

uploadedFiles.push(fileMeta);
fileQueue.push({file:file,fileMeta:fileMeta});
existingNames.add(normalizedFileName);
acceptedCount++;
});

updateFilePanel();

if(skippedMessages.length){
showToast(skippedMessages.join(" | "),"warning");
}

if(acceptedCount>0 && !isProcessing){
processFiles();
}

if(acceptedCount===0 && skippedMessages.length){
showToast("No new valid files added","warning");
}
}

function processFiles(){
if(!fileQueue.length){
isProcessing=false;
currentFileMeta=null;
updateProgress(100);
updateFilePanel();
renderAfterDataChange();

/* Identify any files that failed or were not uploaded */
const issueFiles=uploadedFiles.filter(f=>f.status!=="Uploaded");
const issueNames=issueFiles.map(f=>f.name+" ("+f.status+")");

/* Auto-close Files pop-up after 3 seconds only if all files uploaded successfully */
if(fileDropdown && !issueFiles.length){
setTimeout(()=>{
fileDropdown.classList.remove("active");
},3000);
}

if(issueFiles.length){
showToast("Files processed with issues: "+issueNames.join(", "),"error");
}else{
showToast("All files processed successfully","success");
}

return;
}

isProcessing=true;

const queueItem=fileQueue.shift();
const file=queueItem.file||queueItem;
const ext=getExtension(file.name);

let fileMeta=queueItem.fileMeta;

if(!fileMeta){
fileMeta={
id:"f_"+Date.now()+"_"+Math.random().toString(36).slice(2),
name:file.name,
type:ext,
size:formatSize(file.size),
rawSize:file.size,
status:"Processing",
monthYear:extractMonthYear(file.name),
rowCount:0,
progress:1
};
uploadedFiles.push(fileMeta);
}

fileMeta.status="Processing";
fileMeta.progress=Math.max(fileMeta.progress||0,1);

currentFileMeta=fileMeta;

updateProgress(1);
updateFilePanel();

if(ext==="csv")parseCSV(file,fileMeta);
else parseExcel(file,fileMeta);
}

function parseCSV(file,fileMeta){
updateProgress(5);
const reader=new FileReader();

reader.onprogress=e=>{
if(e.lengthComputable)updateProgress(Math.max(5,Math.floor((e.loaded/e.total)*45)));
};

reader.onload=e=>{
updateProgress(50);
try{
let parsedRows=[];

if(window.Papa){
const result=Papa.parse(e.target.result,{header:true,skipEmptyLines:true});
parsedRows=result.data||[];
if(result.errors&&result.errors.length)showToast("CSV parsed with warnings: "+file.name,"warning");
}else{
parsedRows=fallbackCSVParse(e.target.result);
}

if(!parsedRows.length){
fileMeta.status="Empty";
fileMeta.progress=100;
updateFilePanel();
showToast("Empty CSV file: "+file.name,"warning");
processFiles();
return;
}

updateProgress(70);

const rows=normalizeRows(parsedRows,fileMeta);
fileMeta.rowCount=rows.length;
appendData(rows);
fileMeta.status="Uploaded";
updateFilePanel();

runStableProcessingProgress(rows,()=>processFiles());

}catch(err){
fileMeta.status="Error";
fileMeta.progress=100;
updateFilePanel();
showToast("CSV parsing failed: "+file.name,"error");
processFiles();
}
};

reader.onerror=()=>{
fileMeta.status="Error";
fileMeta.progress=100;
updateFilePanel();
showToast("File read failed: "+file.name,"error");
processFiles();
};

reader.readAsText(file);
}

function parseExcel(file,fileMeta){
updateProgress(5);

if(!window.XLSX){
fileMeta.status="Error";
updateFilePanel();
showToast("SheetJS library not loaded. Failed: "+file.name,"error");
processFiles();
return;
}

const reader=new FileReader();

reader.onprogress=e=>{
if(e.lengthComputable)updateProgress(Math.max(5,Math.floor((e.loaded/e.total)*45)));
};

reader.onload=e=>{
try{
updateProgress(50);

const data=new Uint8Array(e.target.result);
const workbook=XLSX.read(data,{type:"array"});
let allRows=[];

workbook.SheetNames.forEach(sheetName=>{
const sheet=workbook.Sheets[sheetName];
const json=XLSX.utils.sheet_to_json(sheet,{defval:""});
json.forEach(r=>r.__sheetName=sheetName);
allRows=allRows.concat(json);
});

if(!allRows.length){
fileMeta.status="Empty";
fileMeta.progress=100;
updateFilePanel();
showToast("Empty Excel file: "+file.name,"warning");
processFiles();
return;
}

updateProgress(70);

const rows=normalizeRows(allRows,fileMeta);
fileMeta.rowCount=rows.length;
appendData(rows);
fileMeta.status="Uploaded";
updateFilePanel();

runStableProcessingProgress(rows,()=>processFiles());

}catch(err){
fileMeta.status="Error";
fileMeta.progress=100;
updateFilePanel();
showToast("Excel parsing failed: "+file.name,"error");
processFiles();
}
};

reader.onerror=()=>{
fileMeta.status="Error";
fileMeta.progress=100;
updateFilePanel();
showToast("File read failed: "+file.name,"error");
processFiles();
};

reader.readAsArrayBuffer(file);
}

function runStableProcessingProgress(rows,callback){
const total=rows.length||1;

/*
Earlier version used setTimeout chunks only for artificial progress movement.
Browser background tabs may throttle timers, which can delay this artificial progress.
This version completes the post-parse progress immediately and continues queue processing through microtask.
*/
updateProgress(100);

if(currentFileMeta){
currentFileMeta.status="Uploaded";
currentFileMeta.progress=100;
updateFilePanel();
}

const continueQueue=()=>{
if(callback)callback();
};

if(typeof queueMicrotask==="function"){
queueMicrotask(continueQueue);
}else{
Promise.resolve().then(continueQueue);
}
}

function updateProgress(p){
if(percent)percent.innerText=p+"%";
if(progress)progress.style.width=p+"%";

if(currentFileMeta){
currentFileMeta.progress=p;
updateFilePanel();
}
}

function fallbackCSVParse(text){
const lines=text.split(/\r?\n/).filter(Boolean);
if(!lines.length)return[];

const headers=lines[0].split(",").map(h=>h.trim());

return lines.slice(1).map(line=>{
const values=line.split(",");
let obj={};
headers.forEach((h,i)=>obj[h]=values[i]||"");
return obj;
});
}

function buildSectionSummary(sec,data){
const dimension=findColumn(sec.dimensionAliases);
const valueCol=findColumn(sec.valueAliases);
if(!dimension)return{headers:["Message"],rows:[["Required dimension column not detected"]],chartLabels:["No Data"],chartValues:[0],raw:[]};
const map={};
let totalValue=0,totalCount=0;
data.forEach(row=>{
const dim=row[dimension]||"Unknown";
const val=valueCol?toNumber(row[valueCol]):1;
if(!map[dim])map[dim]={label:dim,count:0,amount:0};
map[dim].count++;
map[dim].amount+=val;
totalValue+=val;
totalCount++;
});
let rows=Object.values(map).map(x=>{
const acs=x.count?x.amount/x.count:0;
const share=totalValue?(x.amount/totalValue)*100:(totalCount?(x.count/totalCount)*100:0);
return{label:x.label,count:x.count,amount:x.amount,acs,share};
});
rows.sort((a,b)=>b.amount-a.amount);
rows=rows.slice(0,CONFIG.DEFAULT_TOP_N);
let tableRows=rows.map(r=>{
if(sec.id==="enrollmentAge")return[r.label,formatNumber(r.count),r.share.toFixed(1)+"%"];
if(sec.id==="claimSummary"||sec.id==="disease")return[r.label,formatNumber(r.count),formatCurrency(r.amount),r.share.toFixed(1)+"%"];
if(sec.id==="tat")return[r.label,r.acs.toFixed(1),formatNumber(r.count)];
if(sec.id==="utilization")return[r.label,formatNumber(r.count),formatCurrency(r.amount),r.share.toFixed(1)+"%"];
return[r.label,formatNumber(r.count),formatCurrency(r.amount),formatCurrency(r.acs)];
});
return{
headers:sec.tableColumns,
rows:tableRows,
raw:rows,
chartLabels:rows.map(r=>r.label),
chartValues:rows.map(r=>{
if(sec.id==="enrollmentAge")return r.count;
if(sec.id==="disease")return r.share;
if(sec.id==="siBand"||sec.id==="amountBand")return r.acs;
if(sec.id==="tat")return r.acs;
return r.amount||r.count;
})
};
}

function extractMonthYear(filename){
const name=filename.toLowerCase();
const months={
jan:"Jan",january:"Jan",
feb:"Feb",february:"Feb",
mar:"Mar",march:"Mar",
apr:"Apr",april:"Apr",
may:"May",
jun:"Jun",june:"Jun",
jul:"Jul",july:"Jul",
aug:"Aug",august:"Aug",
sep:"Sep",sept:"Sep",september:"Sep",
oct:"Oct",october:"Oct",
nov:"Nov",november:"Nov",
dec:"Dec",december:"Dec"
};
for(const key in months){
if(name.includes(key)){
const y=name.match(/20\d{2}|\d{2}/);
return months[key]+(y ? "-"+y[0] : "");
}
}
return "";
}

function filterDashboardHomeMenu(){
const input=document.getElementById("homeMenuSearchInput");
const noResult=document.getElementById("homeMenuNoResult");
const query=input ? input.value.trim().toLowerCase() : "";
const categories=document.querySelectorAll(".home-menu-category");

let anyMatch=false;

categories.forEach(category=>{
const categorySearch=(category.getAttribute("data-search")||"").toLowerCase();
const sublist=category.querySelector(".home-menu-sublist");
const items=category.querySelectorAll(".home-menu-item");

let categoryTextMatch=query && categorySearch.includes(query);
let itemMatchCount=0;

items.forEach(item=>{
const itemSearch=(item.getAttribute("data-search")||item.innerText||"").toLowerCase();
const itemMatches=!query || categoryTextMatch || itemSearch.includes(query);

item.style.display=itemMatches ? "flex" : "none";
if(query && itemMatches)itemMatchCount++;
});

const showCategory=!query || categoryTextMatch || itemMatchCount>0;
category.style.display=showCategory ? "" : "none";

if(sublist){
sublist.classList.toggle("force-open",!!query && showCategory);
}

if(showCategory)anyMatch=true;
});

if(noResult)noResult.style.display=anyMatch ? "none" : "block";
}
